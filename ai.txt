// СОСТОЯНИЕ ПРИЛОЖЕНИЯ (ChangeNotifier — нет God-Widget)
// ─────────────────────────────────────────────────────────────────
class AppState extends ChangeNotifier {
  bool _loggedIn = false;
  String _userName = '';
  ThemeMode _themeMode = ThemeMode.light;
  Locale _locale = const Locale('ru');
  bool _notifyClasses = true;
  bool _notifyAnnouncements = true;
  bool _notifyEvents = true;
  // Настраиваемый URL Ollama — можно менять в профиле
  String _ollamaUrl = 'http://localhost:11434';

  bool get loggedIn => _loggedIn;
  String get userName => _userName;
  ThemeMode get themeMode => _themeMode;
  Locale get locale => _locale;
  bool get notifyClasses => _notifyClasses;
  bool get notifyAnnouncements => _notifyAnnouncements;
  bool get notifyEvents => _notifyEvents;
  String get ollamaUrl => _ollamaUrl;

  void login(String name) {
    _loggedIn = true;
    _userName = name;
    notifyListeners();
  }

  void logout() {
    _loggedIn = false;
    _userName = '';
    notifyListeners();
  }

  void setTheme(bool isDark) {
    _themeMode = isDark ? ThemeMode.dark : ThemeMode.light;
    notifyListeners();
  }

  void setLocale(Locale locale) {
    _locale = locale;
    notifyListeners();
  }

  void setNotifyClasses(bool v) { _notifyClasses = v; notifyListeners(); }
  void setNotifyAnnouncements(bool v) { _notifyAnnouncements = v; notifyListeners(); }
  void setNotifyEvents(bool v) { _notifyEvents = v; notifyListeners(); }
  void setOllamaUrl(String url) { _ollamaUrl = url.trim(); notifyListeners(); }
}

// ─────────────────────────────────────────────────────────────────
// InheritedWidget для AppState (минимальный, без Provider-пакета)
// ─────────────────────────────────────────────────────────────────
class _AppStateScope extends InheritedNotifier<AppState> {
  const _AppStateScope({required AppState state, required super.child})
      : super(notifier: state);

  AppState get state => notifier!;

  static AppState of(BuildContext context) =>
      context.dependOnInheritedWidgetOfExactType<_AppStateScope>()!.state;
}

// ─────────────────────────────────────────────────────────────────
// КОРНЕВОЙ ВИДЖЕТ
// ─────────────────────────────────────────────────────────────────
class CollegeDuApp extends StatefulWidget {
  const CollegeDuApp({super.key});

  @override
  State<CollegeDuApp> createState() => _CollegeDuAppState();
}

class _CollegeDuAppState extends State<CollegeDuApp> {
  final _appState = AppState();

  @override
  void dispose() {
    _appState.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return _AppStateScope(
      state: _appState,
      child: ListenableBuilder(
        listenable: _appState,
        builder: (context, _) {
          return MaterialApp(
            title: 'CollegeDu',
            debugShowCheckedModeBanner: false,
            localizationsDelegates: const [
              GlobalMaterialLocalizations.delegate,
              GlobalWidgetsLocalizations.delegate,
              GlobalCupertinoLocalizations.delegate,
            ],
            supportedLocales: const [Locale('ru'), Locale('kk')],
            locale: _appState.locale,
            themeMode: _appState.themeMode,
            theme: ThemeData(
              colorScheme: ColorScheme.fromSeed(seedColor: Colors.indigo),
              useMaterial3: true,
            ),
            darkTheme: ThemeData(
              colorScheme: ColorScheme.fromSeed(
                seedColor: Colors.indigo,
                brightness: Brightness.dark,
              ),
              useMaterial3: true,
            ),
            home: _appState.loggedIn
                ? MainScreen(userName: _appState.userName)
                : const LoginScreen(),
          );
        },
      ),
    );
  }
}

class LoginScreen extends StatefulWidget {
  const LoginScreen({super.key});

  @override
  State<LoginScreen> createState() => _LoginScreenState();
}

class _LoginScreenState extends State<LoginScreen> {
  final _controller = TextEditingController(text: 'Иван Иванов');

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  void _tryLogin() {
    final name = _controller.text.trim();
    final loc = AppL10n.of(context);
    if (name.isEmpty) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text(loc.emptyNameError)),
      );
      return;
    }
    _AppStateScope.of(context).login(name);
  }

  @override
  Widget build(BuildContext context) {
    final loc = AppL10n.of(context);
    return Scaffold(
      body: SafeArea(
        child: SingleChildScrollView(
          padding: EdgeInsets.only(
            left: 24, right: 24, top: 24,
            bottom: MediaQuery.of(context).viewInsets.bottom + 24,
          ),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.stretch,
            children: [
              const SizedBox(height: 48),
              const FlutterLogo(size: 96),
              const SizedBox(height: 24),
              Text(
                loc.welcomeTitle,
                style: const TextStyle(fontSize: 22, fontWeight: FontWeight.bold),
                textAlign: TextAlign.center,
              ),
              const SizedBox(height: 12),
              Text(loc.welcomeSubtitle, textAlign: TextAlign.center),
              const SizedBox(height: 28),
              TextField(
                controller: _controller,
                textInputAction: TextInputAction.done,
                onSubmitted: (_) => _tryLogin(),
                decoration: InputDecoration(
                  labelText: loc.nameLabel,
                  border: const OutlineInputBorder(),
                ),
              ),
              const SizedBox(height: 12),
              ElevatedButton(
                onPressed: _tryLogin,
                child: Text(loc.loginButton),
              ),
              const SizedBox(height: 16),
              Text(
                loc.testDataHint,
                textAlign: TextAlign.center,
                style: const TextStyle(color: Colors.grey),
              ),
              const SizedBox(height: 12),
              Text(
                loc.demoNote,
                textAlign: TextAlign.center,
                style: const TextStyle(color: Colors.grey),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

class MenuPage extends StatefulWidget {
  const MenuPage({super.key});

  @override
  State<MenuPage> createState() => _MenuPageState();
}

class _MenuPageState extends State<MenuPage> {
  String _filter = 'Все';
  final List<MenuItem> _cart = [];

  void _addToCart(MenuItem item) {
    setState(() => _cart.add(item));
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text(AppL10n.of(context).itemAdded(item.name))),
    );
  }

  int get _cartTotal => _cart.fold(0, (sum, item) => sum + item.price);

  @override
  Widget build(BuildContext context) {
    final loc = AppL10n.of(context);
    final types = <String>{'Все', ...AppData.menuItems.map((e) => e.type)};
    final shown = _filter == 'Все'
        ? AppData.menuItems
        : AppData.menuItems.where((m) => m.type == _filter).toList();

    return Scaffold(
      appBar: AppBar(
        title: Text(loc.menuPageTitle),
        actions: [
          Stack(
            alignment: Alignment.center,
            children: [
              IconButton(
                icon: const Icon(Icons.shopping_cart),
                onPressed: () => Navigator.of(context).push(
                  MaterialPageRoute(builder: (_) => PreOrderPage(cart: List.unmodifiable(_cart), total: _cartTotal, onClear: () => setState(() => _cart.clear()))),
                ),
              ),
              if (_cart.isNotEmpty)
                Positioned(
                  right: 6, top: 6,
                  child: CircleAvatar(
                    radius: 8,
                    backgroundColor: Colors.red,
                    child: Text('${_cart.length}', style: const TextStyle(fontSize: 10, color: Colors.white)),
                  ),
                ),
            ],
          ),
        ],
      ),
      body: Padding(
        padding: const EdgeInsets.all(12),
        child: Column(
          children: [
            SizedBox(
              height: 40,
              child: ListView(
                scrollDirection: Axis.horizontal,
                children: types.map((t) => Padding(
                  padding: const EdgeInsets.only(right: 8),
                  child: ChoiceChip(
                    label: Text(t == 'Все' ? loc.allFilter : t),
                    selected: _filter == t,
                    onSelected: (_) => setState(() => _filter = t),
                  ),
                )).toList(),
              ),
            ),
            const SizedBox(height: 12),
            Expanded(
              child: ListView.builder(
                itemCount: shown.length,
                itemBuilder: (_, i) {
                  final m = shown[i];
                  return Card(
                    child: ListTile(
                      title: Text(m.name),
                      subtitle: Text('${m.type} • ${m.price}₸'),
                      trailing: FilledButton.tonal(
                        onPressed: () => _addToCart(m),
                        child: Text(loc.preorderButton),
                      ),
                    ),
                  );
                },
              ),
            ),
          ],
        ),
      ),
    );
  }
}

// AI-АССИСТЕНТ (стриминг-индикатор, настраиваемый URL)
// ─────────────────────────────────────────────────────────────────
class AIAssistantPage extends StatefulWidget {
  const AIAssistantPage({super.key});

  @override
  State<AIAssistantPage> createState() => _AIAssistantPageState();
}

class _AIAssistantPageState extends State<AIAssistantPage> {
  final List<ChatMessage> _messages = [];
  final TextEditingController _controller = TextEditingController();
  final ScrollController _scrollController = ScrollController();
  bool _isLoading = false;

  @override
  void dispose() {
    _controller.dispose();
    _scrollController.dispose();
    super.dispose();
  }

  Future<void> _sendMessage(String text) async {
    final trimmed = text.trim();
    if (trimmed.isEmpty) return;

    setState(() {
      _messages.add(ChatMessage(text: trimmed, isUser: true));
      _isLoading = true;
    });
    _controller.clear();
    _scrollToBottom();

    final loc = AppL10n.of(context);
    final ollamaUrl = _AppStateScope.of(context).ollamaUrl;

    // Проверяем быстрые ответы
    String reply = '';
    for (final qa in loc.quickQA) {
      if (trimmed == qa.question) {
        reply = qa.answer;
        break;
      }
    }

    // Если не нашли — идём в Ollama
    if (reply.isEmpty) {
      reply = await AIAssistantService(trimmed, ollamaUrl: ollamaUrl).getResponse();
    }

    if (!mounted) return; // проверка после await
    setState(() {
      _messages.add(ChatMessage(text: reply, isUser: false));
      _isLoading = false;
    });
    _scrollToBottom();
  }

  void _scrollToBottom() {
    WidgetsBinding.instance.addPostFrameCallback((_) {
      if (_scrollController.hasClients) {
        _scrollController.animateTo(
          _scrollController.position.maxScrollExtent,
          duration: const Duration(milliseconds: 300),
          curve: Curves.easeOut,
        );
      }
    });
  }

  @override
  Widget build(BuildContext context) {
    final loc = AppL10n.of(context);
    return Scaffold(
      appBar: AppBar(title: Text(loc.aiAssistantTitle)),
      body: Column(
        children: [
          Expanded(
            child: _messages.isEmpty
                ? Center(
                    child: Column(
                      mainAxisSize: MainAxisSize.min,
                      children: [
                        Icon(Icons.smart_toy_outlined, size: 72, color: Theme.of(context).colorScheme.primary),
                        const SizedBox(height: 16),
                        Text(loc.aiAssistantTitle, style: const TextStyle(fontSize: 18, fontWeight: FontWeight.bold)),
                        const SizedBox(height: 8),
                        Text(loc.quickQuestions, style: const TextStyle(color: Colors.grey)),
                      ],
                    ),
                  )
                : ListView.builder(
                    controller: _scrollController,
                    padding: const EdgeInsets.all(16),
                    itemCount: _messages.length,
                    itemBuilder: (_, i) => _ChatBubble(message: _messages[i]),
                  ),
          ),
          // Быстрые вопросы
          Container(
            padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 6),
            color: Theme.of(context).colorScheme.surfaceContainerLow,
            child: Wrap(
              spacing: 8,
              runSpacing: 4,
              children: loc.quickQA.map((qa) => ActionChip(
                label: Text(qa.question, overflow: TextOverflow.ellipsis),
                onPressed: () => _sendMessage(qa.question),
              )).toList(),
            ),
          ),
          if (_isLoading)
            Padding(
              padding: const EdgeInsets.symmetric(vertical: 8),
              child: Row(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  const SizedBox(width: 16, height: 16, child: CircularProgressIndicator(strokeWidth: 2)),
                  const SizedBox(width: 12),
                  Text(loc.aiThinking),
                ],
              ),
            ),
          Padding(
            padding: const EdgeInsets.fromLTRB(16, 8, 16, 16),
            child: Row(
              children: [
                Expanded(
                  child: TextField(
                    controller: _controller,
                    textInputAction: TextInputAction.send,
                    onSubmitted: _sendMessage,
                    decoration: InputDecoration(
                      hintText: loc.aiPlaceholder,
                      border: const OutlineInputBorder(),
                      contentPadding: const EdgeInsets.symmetric(horizontal: 12, vertical: 10),
                    ),
                  ),
                ),
                const SizedBox(width: 8),
                IconButton.filled(
                  onPressed: _isLoading ? null : () => _sendMessage(_controller.text),
                  icon: const Icon(Icons.send),
                  tooltip: loc.send,
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }
}

class _ChatBubble extends StatelessWidget {
  final ChatMessage message;
  const _ChatBubble({required this.message});

  @override
  Widget build(BuildContext context) {
    final isUser = message.isUser;
    return Align(
      alignment: isUser ? Alignment.centerRight : Alignment.centerLeft,
      child: Container(
        margin: const EdgeInsets.symmetric(vertical: 4),
        padding: const EdgeInsets.all(12),
        constraints: BoxConstraints(maxWidth: MediaQuery.of(context).size.width * 0.75),
        decoration: BoxDecoration(
          color: isUser ? Theme.of(context).colorScheme.primary : Theme.of(context).colorScheme.surfaceContainerHigh,
          borderRadius: BorderRadius.only(
            topLeft: const Radius.circular(16),
            topRight: const Radius.circular(16),
            bottomLeft: Radius.circular(isUser ? 16 : 0),
            bottomRight: Radius.circular(isUser ? 0 : 16),
          ),
        ),
        child: Row(
          mainAxisSize: MainAxisSize.min,
          crossAxisAlignment: CrossAxisAlignment.end,
          children: [
            if (!isUser) ...[
              CircleAvatar(radius: 14, backgroundColor: Theme.of(context).colorScheme.primary, child: const Icon(Icons.smart_toy, size: 14, color: Colors.white)),
              const SizedBox(width: 8),
            ],
            Flexible(
              child: Text(message.text, style: TextStyle(color: isUser ? Colors.white : Theme.of(context).colorScheme.onSurface)),
            ),
          ],
        ),
      ),
    );
  }
}

/// Виджет настройки URL для Ollama — новая функция
class _OllamaUrlTile extends StatefulWidget {
  final String currentUrl;
  final ValueChanged<String> onSave;
  const _OllamaUrlTile({required this.currentUrl, required this.onSave});

  @override
  State<_OllamaUrlTile> createState() => _OllamaUrlTileState();
}

class _OllamaUrlTileState extends State<_OllamaUrlTile> {
  bool _editing = false;
  late TextEditingController _ctrl;

  @override
  void initState() {
    super.initState();
    _ctrl = TextEditingController(text: widget.currentUrl);
  }

  @override
  void dispose() {
    _ctrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final loc = AppL10n.of(context);
    if (!_editing) {
      return ListTile(
        leading: const Icon(Icons.computer),
        title: Text(loc.ollamaUrlLabel),
        subtitle: Text(widget.currentUrl),
        trailing: IconButton(icon: const Icon(Icons.edit), onPressed: () => setState(() => _editing = true)),
      );
    }
    return Padding(
      padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
      child: Row(
        children: [
          Expanded(
            child: TextField(
              controller: _ctrl,
              decoration: InputDecoration(
                labelText: loc.ollamaUrlLabel,
                hintText: loc.ollamaUrlHint,
                border: const OutlineInputBorder(),
              ),
            ),
          ),
          const SizedBox(width: 8),
          FilledButton(
            onPressed: () {
              widget.onSave(_ctrl.text);
              setState(() => _editing = false);
            },
            child: Text(loc.saveButton),
          ),
        ],
      ),
    );
  }
}
